019import java.io.IOException;
020
021import java.net.URISyntaxException;
022
023import java.nio.file.Files;
024import java.nio.file.Path;
025import java.nio.file.Paths;
026
027import java.util.Map;
028
029import com.fasterxml.jackson.databind.ObjectMapper;
030import com.fasterxml.jackson.databind.MappingIterator;
031import com.fasterxml.jackson.databind.SerializationFeature;
032
033import com.fasterxml.jackson.datatype.jsr310.JavaTimeModule;
034
035import org.junit.Before;
036import org.junit.Test;
037
038import org.microbean.cloudevents.CloudEvent;
039
040import static org.junit.Assert.assertEquals;
041import static org.junit.Assert.assertFalse;
042import static org.junit.Assert.assertNotNull;
043import static org.junit.Assert.assertTrue;
044
045public class TestJsonSerialization {
046
047  private final Path referenceFiles;
048
049  private ObjectMapper objectMapper;
050  
051  public TestJsonSerialization() throws URISyntaxException {
052    super();
053    this.referenceFiles = Paths.get(Thread.currentThread().getContextClassLoader().getResource("TestJsonSerialization/").toURI());
054    assert this.referenceFiles != null;
055  }
056
057  @Before
058  public void createObjectMapper() {
059    this.objectMapper = new ObjectMapper();
060    this.objectMapper.registerModule(new CloudEventsModule());
061    this.objectMapper.registerModule(new JavaTimeModule());
062    this.objectMapper.configure(SerializationFeature.INDENT_OUTPUT, true);
063  }
064
065  @Test
066  public void testRead() throws IOException {
067    final Path testReadJson = this.referenceFiles.resolve("testRead.json");
068    assertNotNull(testReadJson);
069    CloudEvent cloudEvent = null;
070    try (final MappingIterator<CloudEvent> iterator = objectMapper.readerFor(CloudEvent.class).readValues(testReadJson.toUri().toURL())) {
071      assertNotNull(iterator);
072      assertTrue(iterator.hasNext());
073      cloudEvent = iterator.next();
074      assertFalse(iterator.hasNext());
075    }
076    assertNotNull(cloudEvent);
077    assertEquals("0.1", cloudEvent.getCloudEventsVersion());
078    assertEquals("com.example.someevent", cloudEvent.getEventType());
079    assertEquals("1.0", cloudEvent.getEventTypeVersion());
080    assertEquals("/mycontext", String.valueOf(cloudEvent.getSource()));
081    assertEquals("A234-1234-1234", cloudEvent.getEventID());
082    assertNotNull(cloudEvent.getEventTime()); // TODO: verify parsing
083    final Map<? extends String, ?> extensions = cloudEvent.getExtensions();
084    assertNotNull(extensions);
085    assertFalse(extensions.isEmpty());
086    assertEquals(1, extensions.size());
087    assertEquals("value", extensions.get("comExampleExtension"));
088    assertEquals("text/xml", cloudEvent.getContentType());
089    assertEquals("<much wow=\"xml\"/>", cloudEvent.getData());
090  }
091  
092}
